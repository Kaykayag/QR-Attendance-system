        function getFromStorage(key, defaultValue = []) {
            try {
                const data = localStorage.getItem(key);
                return data ? JSON.parse(data) : defaultValue;
            } catch (e) {
                console.error('Error reading from storage:', e);
                return defaultValue;
            }
        }

        function saveToStorage(key, data) {
            try {
                localStorage.setItem(key, JSON.stringify(data));
            } catch (e) {
                console.error('Error saving to storage:', e);
            }
        }

        // ============ QR SCANNING ============
        function toggleScanner() {

            if (wrapper.classList.contains('hidden')) {

        function handleImageUpload(event) {
            const file = event.target.files[0];
            if (!file) return;
            
            // Validate file type
            if (!file.type.startsWith('image/')) {
                alert('‚ùå Please select a valid image file.');
                event.target.value = '';
                return;
            }
            
            // Validate file size (max 5MB)
            if (file.size > 5 * 1024 * 1024) {
                alert('‚ùå Image file is too large. Please select an image smaller than 5MB.');
                event.target.value = '';
                return;
            }
            
            alert('üì± QR Image uploaded! Decoding...');
            
            try {
                const img = new Image();
                const canvas = document.createElement('canvas');
                const ctx = canvas.getContext('2d');
                
                img.onload = function() {
                    try {
                        canvas.width = img.width;
                        canvas.height = img.height;
                        ctx.drawImage(img, 0, 0);
                        
                        const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
                        
                        const code = jsQR(imageData.data, imageData.width, imageData.height, {
                            inversionAttempts: "dontInvert",
                        });
                        
                        if (code) {
                            console.log('QR Code decoded from image:', code.data);
                            processQRCode(code.data);
                        } else {
                            alert('‚ùå No QR code found in the image. Please ensure:\n' +
                                  '‚Ä¢ The image is clear and well-lit\n' +
                                  '‚Ä¢ The QR code is not blurry or distorted\n' +
                                  '‚Ä¢ The QR code is fully visible in the image\n' +
                                  '‚Ä¢ Try a different QR code if available');
                        }
                    } catch (decodeError) {
                        console.error('Error decoding QR code:', decodeError);
                        alert('‚ùå Error processing QR code. Please try a different image.');
                    }
                };
                
                img.onerror = function() {
                    alert('‚ùå Error loading image. Please try a different image.');
                };
                
                const reader = new FileReader();
                reader.onload = function(e) {
                    img.src = e.target.result;
                };
                reader.onerror = function() {
                    alert('‚ùå Error reading file. Please try again.');
                };
                reader.readAsDataURL(file);
                
            } catch (error) {
                console.error('Error processing image:', error);
                alert('‚ùå Error processing image. Please try again.');
            }
            
            event.target.value = '';
        }

        function processQRCode(qrData) {
            // Extract student ID from QR data
            const studentId = extractStudentIdFromQR(qrData);
            
            if (!studentId) {
                alert('‚ùå Invalid QR code format. Expected STUDENT:ID format.');
                return;
            }
            
            // Find student
            const students = getFromStorage(STORAGE_KEYS.STUDENTS);
            const student = students.find(s => s.id === studentId && s.isActive);
            
            if (!student) {
                alert(`‚ùå Student ID ${studentId} not found.`);
                return;
            }
            
            // Mark attendance
            markAttendance(student);
        }

        function extractStudentIdFromQR(qrData) {
            if (!qrData) return null;
            
            // Try to extract from "STUDENT:ID" format
            if (qrData.startsWith('STUDENT:')) {
                return qrData.split('STUDENT:')[1];
            }
            
            // Try to extract numeric ID directly
            const numericMatch = qrData.match(/\d+/);
            if (numericMatch) {
                return numericMatch.group();
            }
            
            return null;
        }

        function markAttendance(student) {
            const attendanceLogs = getFromStorage(STORAGE_KEYS.ATTENDANCE);
            const today = new Date().toDateString();
            
            // Check if already marked today
            const existingLog = attendanceLogs.find(log => 
                log.studentId === student.id && 
                new Date(log.date).toDateString() === today
            );
            
            if (existingLog) {
                alert(`‚ö†Ô∏è ${student.firstName} ${student.lastName} is already marked present today.`);
                return;
            }
            
            // Create attendance log
            const logEntry = {
                id: Date.now().toString(),
                studentId: student.id,
                studentName: `${student.firstName} ${student.lastName}`,
                courseLevel: `${student.course} - Year ${student.level}`,
                date: new Date().toISOString(),
                time: new Date().toLocaleTimeString(),
                teacherId: currentTeacher ? currentTeacher.id : '',
                teacherName: currentTeacher ? `${currentTeacher.firstName} ${currentTeacher.lastName}` : '',
                method: 'QR_SCAN'
            };
            
            attendanceLogs.unshift(logEntry);
            saveToStorage(STORAGE_KEYS.ATTENDANCE, attendanceLogs);
            
            // Update UI
            addAttendanceToTable(logEntry);
            presentCount++;
            document.getElementById('present-count').innerText = presentCount;
            updateAttendanceRate();
            
            alert(`‚úÖ ${student.firstName} ${student.lastName} marked present!`);
        }

        function addAttendanceToTable(logEntry) {
            const tbody = document.getElementById('attendance-list');
            const empty = document.getElementById('empty-state');
            if (empty) empty.remove();
            
            const row = `
                <tr class="hover:bg-green-50 transition bg-green-100 text-green-800">
                    <td class="px-6 py-4 text-gray-500 font-mono text-sm">${logEntry.time}</td>
                    <td class="px-6 py-4 font-medium text-gray-900">
                        ${logEntry.studentName} <div class="text-xs text-gray-400">${logEntry.courseLevel}</div>
                    </td>
                    <td class="px-6 py-4"><span class="bg-green-100 text-green-800 px-2 py-1 rounded-full text-xs">Present</span></td>
                </tr>`;
            tbody.innerHTML = row + tbody.innerHTML;
        }

        // ============ STUDENT MANAGEMENT ============
        function showStudentsManagement() {

        function loadStudentsList() {
            currentStudents = getFromStorage(STORAGE_KEYS.STUDENTS).filter(s => s.isActive);
            renderStudentsTable(currentStudents);

        function renderStudentsTable(students) {
            const tbody = document.getElementById('students-table-body');
            
            if (students.length === 0) {
                tbody.innerHTML = '<tr><td colspan="5" class="px-6 py-12 text-center text-gray-400">No students found</td></tr>';
                return;
            }
            
            // Get today's attendance to check status
            const attendanceLogs = getFromStorage(STORAGE_KEYS.ATTENDANCE);
            const today = new Date().toDateString();
            
            tbody.innerHTML = students.map(student => {
                const isPresentToday = attendanceLogs.some(log => 
                    log.studentId === student.id && 
                    new Date(log.date).toDateString() === today
                );
                
                return `
                <tr class="hover:bg-gray-50">
                    <td class="px-6 py-4 font-mono text-sm">${student.id}</td>
                    <td class="px-6 py-4 font-medium">${student.firstName} ${student.lastName}</td>
                    <td class="px-6 py-4 text-gray-600">${student.course} - Year ${student.level}</td>
                    <td class="px-6 py-4">
                        <span class="${isPresentToday ? 'bg-green-100 text-green-800' : 'bg-gray-100 text-gray-800'} px-2 py-1 rounded-full text-xs">
                            ${isPresentToday ? 'Present Today' : 'Absent'}
                        </span>
                    </td>
                    <td class="px-6 py-4">
                        <div class="flex space-x-2">
                            <button onclick="editStudent('${student.id}')" class="text-blue-600 hover:text-blue-800 text-sm font-medium">
                                <i data-lucide="edit" class="w-4 h-4"></i>
                            </button>
                            <button onclick="downloadStudentQR('${student.id}')" class="text-green-600 hover:text-green-800 text-sm font-medium">
                                <i data-lucide="download" class="w-4 h-4"></i>
                            </button>
                            <button onclick="deleteStudent('${student.id}')" class="text-red-600 hover:text-red-800 text-sm font-medium">
                                <i data-lucide="trash-2" class="w-4 h-4"></i>
                            </button>
                        </div>
                    </td>
                </tr>`;
            }).join('');
            
            lucide.createIcons();
        }

        function refreshStudentsList() {
            loadStudentsList();

        function searchStudents() {
            const searchTerm = document.getElementById('student-search').value.toLowerCase();
            const filteredStudents = currentStudents.filter(student => 
                `${student.firstName} ${student.lastName}`.toLowerCase().includes(searchTerm) ||
                student.id.toLowerCase().includes(searchTerm) ||
                student.course.toLowerCase().includes(searchTerm)
            );
            renderStudentsTable(filteredStudents);
        }

        function editStudent(studentId) {
            const students = getFromStorage(STORAGE_KEYS.STUDENTS);
            const student = students.find(s => s.id === studentId);
            if (!student) return;
            
            const newName = prompt('Enter new name (First Last):', `${student.firstName} ${student.lastName}`);
            if (!newName) return;
            
            const [firstName, ...lastNameParts] = newName.split(' ');
            const lastName = lastNameParts.join(' ');
            
            const newCourse = prompt('Enter new course:', student.course);
            if (!newCourse) return;
            
            const newLevel = prompt('Enter new level (1-5):', student.level);
            if (!newLevel) return;
            
            student.firstName = firstName;
            student.lastName = lastName;
            student.course = newCourse;
            student.level = newLevel;
            
            saveToStorage(STORAGE_KEYS.STUDENTS, students);
            loadStudentsList();
            alert('‚úÖ Student updated successfully!');
        }

        function deleteStudent(studentId) {
            if (!confirm('Are you sure you want to deactivate this student?')) return;
            
            const students = getFromStorage(STORAGE_KEYS.STUDENTS);
            const studentIndex = students.findIndex(s => s.id === studentId);
            
            if (studentIndex >= 0) {
                students[studentIndex].isActive = false;
                saveToStorage(STORAGE_KEYS.STUDENTS, students);
                loadStudentsList();
                alert('‚úÖ Student deactivated successfully!');
            }
        }

        function downloadStudentQR(studentId) {
            const students = getFromStorage(STORAGE_KEYS.STUDENTS);
            const student = students.find(s => s.id === studentId);
            if (!student) return;
            
            const qrData = `STUDENT:${studentId}`;
            const qrUrl = `https://api.qrserver.com/v1/create-qr-code/?size=300x300&data=${encodeURIComponent(qrData)}`;
            
            const link = document.createElement('a');
            link.href = qrUrl;
            link.download = `${student.firstName}_${student.lastName}_QR.png`;
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }

        // ============ DASHBOARD STATS ============
        function loadDashboardStats() {
            const students = getFromStorage(STORAGE_KEYS.STUDENTS).filter(s => s.isActive);
            const attendanceLogs = getFromStorage(STORAGE_KEYS.ATTENDANCE);
            const today = new Date().toDateString();
            
            const totalStudents = students.length;
            const presentToday = attendanceLogs.filter(log => 
                new Date(log.date).toDateString() === today
            ).length;
            
            presentCount = presentToday;
            document.getElementById('present-count').innerText = presentToday;
            document.getElementById('total-students').innerText = totalStudents;
            
            const attendanceRate = totalStudents > 0 ? Math.round((presentToday / totalStudents) * 100) : 0;
            document.getElementById('attendance-rate').innerText = `${attendanceRate}%`;
        }
